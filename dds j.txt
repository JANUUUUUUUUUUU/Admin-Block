<script>
  // Sample fees data by course and year
  const feesData = {
    btech: {
      1: { tuition: 60000, hostel: 25000, transport: 5000, misc: 3000 },
      2: { tuition: 60000, hostel: 25000, transport: 5000, misc: 3000 },
      3: { tuition: 65000, hostel: 26000, transport: 5500, misc: 3500 },
      4: { tuition: 65000, hostel: 26000, transport: 5500, misc: 3500 }
    },
    mba: {
      1: { tuition: 75000, hostel: 30000, transport: 6000, misc: 4000 },
      2: { tuition: 75000, hostel: 30000, transport: 6000, misc: 4000 },
      3: { tuition: 0, hostel: 0, transport: 0, misc: 0 }, // MBA usually 2 years
      4: { tuition: 0, hostel: 0, transport: 0, misc: 0 }
    },
    mca: {
      1: { tuition: 50000, hostel: 22000, transport: 4000, misc: 2500 },
      2: { tuition: 50000, hostel: 22000, transport: 4000, misc: 2500 },
      3: { tuition: 52000, hostel: 23000, transport: 4200, misc: 2800 },
      4: { tuition: 0, hostel: 0, transport: 0, misc: 0 } // MCA usually 3 years
    }
  };

  document.getElementById('studentForm').addEventListener('submit', function(e) {
    e.preventDefault();

    const name = document.getElementById('studentName').value.trim();
    const course = document.getElementById('course').value;
    const year = document.getElementById('year').value;

    // Check if fees data exists for the selected course/year
    const fees = feesData[course][year];
    if(!fees || (fees.tuition === 0 && fees.hostel === 0)) {
      alert("No fees data available for this course/year.");
      return;
    }

    const total = fees.tuition + fees.hostel + fees.transport + fees.misc;

    // Update the UI
    document.getElementById('studentNameDisplay').textContent = name;
    document.getElementById('tuitionFee').textContent = `₹${fees.tuition.toLocaleString()}`;
    document.getElementById('hostelFee').textContent = `₹${fees.hostel.toLocaleString()}`;
    document.getElementById('transportFee').textContent = `₹${fees.transport.toLocaleString()}`;
    document.getElementById('miscFee').textContent = `₹${fees.misc.toLocaleString()}`;
    document.getElementById('totalFee').textContent = `₹${total.toLocaleString()}`;

    document.getElementById('feesDetails').style.display = 'block';
  });
</script>
